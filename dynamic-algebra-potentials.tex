\title{Dynamic Algebra of Potentials}

\author{David I. Spivak}

\date{\vspace{-.2in}}

\maketitle

\begin{abstract}
We construct a symmetric monoidal functor $F\colon\plenspot\to\org$ from a category of potential lenses between smooth manifolds into the Shapiro--Spivak monoidal bicategory of polynomial coalgebras. The state of each $F(\ell)$ is the cotangent bundle of a SCAN parameter group, and the state-update is Hamilton's equations on that bundle, decomposed into force-like contributions from the lens's data. The construction and its functoriality are structural consequences of three general facts about strong symmetric monoidal functors from a cartesian category into a symmetric monoidal closed category, specialized to the cotangent functor $\cot\colon\mfd\to\poly$.
\end{abstract}

\section{Polynomial functors, comonads, and coalgebras}\label{sec:poly-dynamic-bg}

\subsection{The symmetric monoidal closed category $\poly$}\label{sec:poly}

We briefly review the category $\poly$ of polynomial functors; see~\cite{niu2025polynomial} for a comprehensive treatment and~\cite{spivak2026changing} for a concise reference.

A \emph{polynomial functor} is an endofunctor $p\colon\smset\to\smset$ of the form $\sum_{i: I}\yon^{A_i}$, where $I$ is the set of \emph{positions} and $A_i$ is the set of \emph{directions} at position $i$. Since $p(1)=\sum_{i:I}1=I$, we use the compact notation $p(1)$ for the position set and $p[i]\coloneqq A_i$ for the direction set at $i: p(1)$, so that
\begin{equation}\label{eqn:poly}
p=\sum_{i: p(1)}\yon^{p[i]}=\sum_{i: p(1)}\;\prod_{d: p[i]}\yon.
\end{equation}
We use juxtaposition for scalar multiplication: $pq\coloneqq p\times q$, and we write $1=\yon^0$. A \emph{monomial} is a polynomial of the form $M\yon^N=\sum_{m:M}\yon^N$; we refer to $\yon^N$ as a \emph{representable} and to $M=M\yon^0$ as a \emph{constant}.

One can derive from the universal property of coproducts and the Yoneda lemma that natural transformations $\varphi\colon p\to q$ are given by:
\begin{equation}\label{eqn:poly_map}
\poly(p,q)=\prod_{i: p(1)}\;\sum_{j: q(1)}\;\prod_{e: q[j]}\; \sum_{d: p[i]}1.
\end{equation}
The outer $\prod_i\sum_j$ yields an ``forward on-positions'' function $\varphi_1\colon p(1)\to q(1)$, which is the component of $\varphi$ at $1:\smset$, and the inner $\prod_e\sum_d$ yields, for each $i:p(1)$, a ``backward on-directions'' function from the directions of $q$ at $\varphi_1(i)$ to the directions of $p$ at $i$; we denote this function
\[\bk{\varphi}_i\colon q[\varphi_1(i)]\to p[i],\]
writing the overline-bracket $\bk{(\,\cdot\,)}$ for the backward component throughout. We take the morphisms $p\to q$ in $\poly$ to be the natural transformations between the underlying functors.

For polynomials $p,q$, their \emph{Dirichlet tensor product} is
\begin{equation}\label{eqn:dirichlet}
p\otimes q\coloneqq\sum_{(i,j):p(1)\times q(1)}\yon^{p[i]\times q[j]}\;.
\end{equation}
The unit is $\yon=\yon^1$. The Dirichlet product is symmetric monoidal and closed: the \emph{internal hom} is
\begin{equation}\label{eqn:dirichlet_hom}
\ihom{p,q}=\prod_{i: p(1)}\;\sum_{j: q(1)}\;\prod_{e: q[j]}\;\sum_{d: p[i]}\yon
\end{equation}
satisfying $\poly(r\otimes p,\, q)\cong\poly(r,\,\ihom{p,q})$ naturally in $r$. Note that $\ihom{p,q}(1)=\poly(p,q)$, as is immediate from comparing \eqref{eqn:dirichlet_hom} with \eqref{eqn:poly_map}. We can rewrite it as
\[
\ihom{p,q}\cong\sum_{\varphi: \poly(p,q)}\yon^{\sum_{i: p(1)}q[\varphi_1(i)]}
\]
A direction at $\varphi:\ihom{p,q}(1)$ consists of a pair $(i,e)$, where $i:p(1)$ is a $p$-position and $e:q[\varphi_1(i)]$ is a $q$-direction.

\subsection{Polynomial comonads as categories}\label{sec:comonads}

The \emph{substitution} (or \emph{composition}) \emph{product} on $\poly$ is defined by $(p\tri q)(S)\coloneqq p(q(S))$, with unit $\yon$. Using the expanded form of \eqref{eqn:poly}:
\begin{equation}\label{eqn:substitution}
p\tri q=\sum_{i: p(1)}\;\prod_{d: p[i]}\;\sum_{j: q(1)}\;\prod_{e: q[j]}\yon.
\end{equation}
A position of $p\tri q$ is a pair $(i,j)$ where $i: p(1)$ and $j\colon p[i]\to q(1)$; the direction set is $(p\tri q)[(i,j)]=\sum_{d:p[i]}q[j_d]$. In particular, viewing a set $S:\smset$ as the constant polynomial $S\yon^0$, substitution recovers evaluation: $p\tri S=p(S)$.

A \emph{$\tri$-comonoid} in $\poly$ is a comonoid for the substitution product: a polynomial $c$ equipped with a counit $\epsilon\colon c\to\yon$ and comultiplication $\delta\colon c\to c\tri c$ satisfying coassociativity and counitality. Since the substitution product is composition of endofunctors on $\smset$, a $\tri$-comonoid is equivalently a comonad on $\smset$ whose underlying endofunctor is polynomial; we call it a \emph{polynomial comonad}.

Polynomial comonads are---up to isomorphism---small categories. Writing $c=\sum_{i:c(1)}\yon^{c[i]}$, the positions $c(1)$ are the objects of some category $\cat{C}$, and for each object $i:c(1)$, the directions $c[i]$ are the morphisms of $\cat{C}$ with domain $i$:
\[
c[i]\coloneqq\sum_{j:c(1)}\cat{C}(i,j).
\]
The counit $\epsilon\colon c\to\yon$ picks out an identity morphism $\id_i:c[i]$ at each object $i$. The comultiplication $\delta\colon c\to c\tri c$ assigns codomains and composition: given an object $i$ and a morphism $f:c[i]$, the on-positions component of $\delta$ returns $\cod(f):c(1)$, and given a further morphism $g:c[\cod(f)]$, the on-directions component returns the composite $f\then g:c[i]$. The comonoid laws guarantee that $\cod(\id_i)=i$, that composition is unital, that $\cod(f\then g)=\cod(g)$, and that composition is associative.

\begin{proposition}[Polynomial comonads are categories {\cite{ahman2016directed}; see also \cite{niu2025polynomial}}]\label{prop:comonads-cats}
The above correspondence defines an isomorphism between polynomial comonads and small categories.
\end{proposition}

\begin{example}[The store comonad and the pair groupoid]\label{ex:store}
For any set $S$, the monomial $S\yon^S$ arises as the store comonad from the adjunction $S\times-\;\dashv\;(-)^S$ on $\smset$, i.e.\ the comonad with underlying endofunctor $X\mapsto S\times X^S$. We can also write
\begin{equation}\label{eqn:store-decomp}
S\yon^S\;\cong\;\ihom{S\yon,\yon}\otimes S\yon
\end{equation}
since $\ihom{S\yon,\yon}=\yon^S$, and $\yon^S\otimes S\yon\cong S\yon^{S\times 1}\cong S\yon^S$. The counit $\epsilon\colon S\yon^S\to\yon$ sends each position $s:S$ to the identity $\id_s\coloneqq s:S=c[s]$. The comultiplication $\delta\colon S\yon^S\to S\yon^S\tri S\yon^S$ acts as follows: on positions, $\delta$ sends $s:S$ to the pair $(s,\cod)$ where $\cod\colon S\to S$ is the identity---so $\cod(t)=t$ for all $t:S$. On directions, given $s:S$ and $t:c[s]=S$, the codomain is $\cod(t)=t$, and given a further $u:c[t]=S$, the composite is $t\then u\coloneqq u$.

By \cref{prop:comonads-cats} this is a category: it has object set $S$ and, for each pair $(s,t):S\times S$, exactly one morphism $s\to t$. The composition $t\then u=u$ says that composites are determined by the codomain of the second factor, as expected when every hom-set is a singleton. Every arrow $s\to t$ is an isomorphism with inverse $t\to s$, so this is the \emph{pair groupoid} (or \emph{codiscrete groupoid}) on $S$.
\end{example}


\begin{proposition}[{\cite{niu2025polynomial}}]\label{prop:lens}
$\poly$ has a left Kan operation $\lens{p}{q}$ with
\[
\poly(p,q'\tri q)\cong\poly\left(\lens{p}{q},q'\right)
\]
given by the formula
\begin{equation}\label{eqn.lens}
\lens{p}{q}=\sum_{i:p(1)}\yon^{q\tri p[i]}.
\end{equation}
\end{proposition}


\subsection{Polynomial coalgebras as dynamical systems}\label{sec:coalgebras}

For a polynomial $p$, a \emph{$p$-coalgebra} is a pair $(S,\beta)$ where $S:\smset$ is a set and $\beta\colon S\to p(S)$ is a function. We call elements of $S$ states; given a state $s:S$, the map assigns a \emph{readout} position $r_s\coloneqq\beta_1(s):p(1)$ and an \emph{update} function $\bk{\beta}_s\colon p[r_s]\to S$ sending each direction at $r_s$ to a new state. For example when $p=A\yon^B$ then a $p$-coalgebra $S\to AS^B$ is equivalently a readout function $S\to A$ and an update function $S\times B\to S$.

A $p$-coalgebra is thus a deterministic dynamical system whose interface is $p$: the state exposes an output in $p(1)$ and transitions in response to inputs in $p[\beta_1(s)]$, capturing Moore machines, discretized differential equations, Markov decision processes, and other mode-dependent dynamics in a single framework~\cite{spivak2020poly,niu2025polynomial}. It is a special case of the general theory of coalgebras for endofunctors; see~\cite{jacobs2017introduction} for a comprehensive treatment.

\begin{example}[Coalgebras as polynomial maps]\label{ex:coalg-as-poly-map}
A $p$-coalgebra $\beta\colon S\to p\tri S$ can be identified via \cref{prop:lens} as a map $\lens{S}{S}\to p$. By the formula \eqref{eqn.lens}, $\lens{S}{S}=S\yon^S$, so $\beta$ can be rewritten as a map
\[
\beta'\colon S\yon^S\to p.
\qedhere\]
\end{example}



\section{Manifolds, Lie groups, and SCANs}\label{sec:lie-scan-bg}

\subsection{The cotangent functor}\label{sec:cot}

The category of smooth manifolds provides the source of polynomials relevant to physics. Each manifold has a cotangent bundle, and each smooth map has a Jacobian transpose; combining these gives a functor into $\poly$. We follow the treatment in~\cite{spivak2026dynprop}.

\begin{definition}[Cotangent functor]\label{def:cot}
The \emph{cotangent functor} $\cot\colon\mfd\to\poly$ sends a smooth manifold $M$ to
\begin{equation}\label{eqn:cot_def}
\cot(M)\coloneqq\sum_{m:M}\yon^{T^*_mM}
\end{equation}
where $T^*_mM$ is the cotangent space at $m$. A smooth map $f\colon M\to N$ is sent to the $\poly$ map $\cot(f)\colon\cot(M)\to\cot(N)$ with forward part $f$ and backward part $(Df_m)^\top\colon T^*_{f(m)}N\to T^*_mM$ (the Jacobian transpose at $m$).
\end{definition}

\begin{proposition}\label{prop:cot-monoidal}
The cotangent functor $\cot\colon(\mfd,\mathbb{R}^0,\times)\to(\poly,\yon,\otimes)$ is strong symmetric monoidal:
\[
\cot(\mathbb{R}^0)\cong\yon
\qqand
\cot(M\times N)\cong\cot(M)\otimes\cot(N).
\]
\end{proposition}

\begin{proof}
\textit{Functoriality.} The identity map $\id_M\colon M\to M$ has Jacobian $D(\id_M)_m=\id_{T_mM}$, whose transpose is $\id_{T^*_mM}$, so $\cot(\id_M)=\id_{\cot(M)}$. For composable smooth maps $f\colon M\to N$ and $g\colon N\to P$, the chain rule gives $D(g\circ f)_m=Dg_{f(m)}\circ Df_m$, so
\[
(D(g\circ f)_m)^\top=(Df_m)^\top\circ(Dg_{f(m)})^\top=\bk{\cot(f)}_m\circ\bk{\cot(g)}_{f(m)},
\]
which is the backward part of $\cot(f)\then\cot(g)$. (Note: $\cot$ is covariant precisely because it takes Jacobian \emph{transposes}, which reverses the backward direction.)

\textit{Unit.} The one-point manifold $\mathbb{R}^0=\{*\}$ has $T^*_*\mathbb{R}^0=0$, so $\cot(\mathbb{R}^0)=\sum_{*:1}\yon^0=\yon$.

\textit{Products.} The cotangent space of a product splits: $T^*_{(m,n)}(M\times N)\cong T^*_mM\times T^*_nN$. Thus
\begin{equation}\label{eqn.cot_times}
\cot(M\times N)=\sum_{(m,n):M\times N}\yon^{T^*_mM\times T^*_nN}=\cot(M)\otimes\cot(N)
\end{equation}
using the definition of the Dirichlet product \eqref{eqn:dirichlet}.

\textit{Naturality.} For a product map $(f,g)\colon M\times N\to M'\times N'$, the Jacobian is block-diagonal: $D(f,g)_{(m,n)}=Df_m\oplus Dg_n$. Hence $(D(f,g)_{(m,n)})^\top=(Df_m)^\top\times(Dg_n)^\top$, so the isomorphism $\cot(M\times N)\cong\cot(M)\otimes\cot(N)$ is natural in $M$ and~$N$.

\textit{Coherence.} The associator and unitor for the Dirichlet product correspond to the canonical isomorphisms $(M\times N)\times P\cong M\times(N\times P)$ and $M\times 1\cong M$ in $\mfd$, so the coherence diagrams for a strong monoidal functor commute.
\end{proof}

\Cref{prop:cot-monoidal} uses only the underlying sets of the cotangent fibers, e.g.\ in \eqref{eqn.cot_times}. The linear structure is not lost, however: \cref{lem:cot-comonoid} shows that the diagonal and terminal maps in $\mfd$ induce fiberwise addition and zero on $\cot(M)$.

\begin{lemma}\label{lem:cot-comonoid}
For any smooth manifold $M$, the polynomial $\cot(M)$ is a cocommutative comonoid in $(\poly,\yon,\otimes)$. The comultiplication and counit
\[
\cot(\Delta_M)\colon\cot(M)\to\cot(M)\otimes\cot(M)
\qqand
 \cot(!_M)\colon\cot(M)\to\yon
\]
sends each position $m:M$ to $(m,m)$. On directions, each cotangent fibre $T^*_mM$ carries the induced cocommutative comonoid structure
\[
T^*_mM\times T^*_mM\to T^*_mM,\quad (\xi,\eta)\mapsto\xi+\eta
\qqand 1\to T^*_mM,\quad *\mapsto 0,
\]
i.e.\ fiberwise addition and zero. Thus the $\otimes$-comonoid structure on $\cot(M)$ recovers the additive structure of each cotangent space.
\end{lemma}

\begin{proof}
Since $(\mfd,\mathbb{R}^0,\times)$ is cartesian, every manifold $M$ is a cocommutative comonoid via the diagonal $\Delta_M\colon M\to M\times M$ and $!_M\colon M\to\mathbb{R}^0$. Since $\cot$ is strong symmetric monoidal (\cref{prop:cot-monoidal}), it preserves comonoid structures. On positions this gives $m\mapsto(m,m)$ and $m\mapsto *$. For the backward maps: the Jacobian of $\Delta_M$ at $m$ is the diagonal $T_mM\to T_mM\times T_mM$, $v\mapsto(v,v)$; its transpose is addition $T^*_mM\times T^*_mM\to T^*_mM$. The Jacobian of $!_M$ at $m$ is $T_mM\to 0$, whose transpose is the map $1\to T^*_mM$ that selects the zero covector.
\end{proof}

\begin{proposition}[Hopf monoid structure on cotangent bundles of Lie groups]\label{prop:cot-hopf}
Let $G$ be a Lie group with unit $\eta\colon\mathbb{R}^0\to G$, multiplication $\mu\colon G\times G\to G$, and inversion $\iota\colon G\to G$. Then $\cot(G)$ is a Hopf monoid in $(\poly,\yon,\otimes)$: the monoid structure is
\[
\cot(\eta)\colon\yon\to\cot(G),\qquad \cot(\mu)\colon\cot(G)\otimes\cot(G)\to\cot(G),
\]
the comonoid structure is as in \cref{lem:cot-comonoid}, and the antipode is $\cot(\iota)\colon\cot(G)\to\cot(G)$.
\end{proposition}

\begin{proof}
The Lie group axioms are equations in $\mfd$; applying the strong monoidal functor $\cot$ preserves them. In particular, the monoid axioms for $(\eta,\mu)$ become monoid axioms for $(\cot(\eta),\cot(\mu))$, the compatibility between multiplication and comultiplication (which holds because $\Delta$ is natural), and the antipode condition $\mu\circ(\iota\times\id)\circ\Delta=\eta\circ!=\mu\circ(\id\times\iota)\circ\Delta$ maps to the Hopf monoid axiom in $\poly$. See also~\cite{spivak2023hopf}.
\end{proof}

\subsection{Lie groups and notation}\label{sec:lie-groups}

Recall that a Lie group is a group object in $\mfd$, i.e.\ a manifold $G$ equipped with a unit $\eta\colon \mathbb{R}^0\to G$, multiplication $\mu\colon G\times G\to G$, and inversion $\iota\colon G\to G$. We write
\[
e\coloneqq \eta(*):G,\qquad gh\coloneqq \mu(g,h),\qquad g\inv\coloneqq \iota(g).
\]
The \emph{Lie algebra of $G$} is the tangent space at the identity:
\[
\lie{g}\coloneqq T_eG.
\]
For each $g:G$, left translation is the diffeomorphism
\[
L_g\colon G\to G,\qquad h\mapsto gh.
\]
Its differential at the identity is a linear isomorphism and we denote it and its transpose by
\[
dL_g|_e\colon \lie{g}\to T_gG
\qqand
(dL_g|_e)^*\colon T^*_gG\to \lie{g}^*.
\]
Thus every cotangent fiber $T^*_gG$ is canonically identified with the single vector space $\lie{g}^*$ by pulling covectors back along left translation. Collecting these fiberwise identifications gives a vector-bundle isomorphism over $G$:
\begin{equation}\label{eqn.left_triv}
T^*G \cong G\times\lie{g}^*,\qquad \xi\mapsto (g,(dL_g|_e)^*\xi)\quad\text{for }\xi:T^*_gG.
\end{equation}
We refer to this bundle isomorphism as the \emph{left-trivialization} of $T^*G$.

Conjugation by $g$ is the Lie group automorphism
\[
c_g\colon G\to G,\qquad h\mapsto ghg\inv.
\]
Its differential at the identity is the adjoint action, and the transpose of the adjoint action is the coadjoint action:
\[
\Ad_g\coloneqq dc_g|_e\colon \lie{g}\to \lie{g}
\qqand
\Ad_g^*\coloneqq (dc_g|_e)^*\colon \lie{g}^*\to \lie{g}^*.
\]

\begin{lemma}\label{lem:Ad-facts}
The adjoint and coadjoint actions define group homomorphisms
\[
\Ad\colon G\to \Aut(\lie{g})
\qqand
\Ad^*\colon G\op\to \Aut(\lie{g}^*).
\]
\end{lemma}

\begin{proof}
To say that $\Ad$ is a group homomorphism is exactly to say that it preserves the unit and multiplication. Unwinding this gives:
\[
\Ad_e=\id_{\lie{g}}
\qqand
\Ad_{gh}=\Ad_g\circ \Ad_h.
\]
Indeed, $c_e=\id_G$, so $\Ad_e=dc_e|_e=\id_{\lie{g}}$, and $c_{gh}=c_g\circ c_h$ for all $g,h:G$, so differentiating at $e$ gives $\Ad_{gh}=\Ad_g\circ \Ad_h$. Taking transposes gives
\[
\Ad_e^*=\id_{\lie{g}^*}
\qqand
\Ad_{gh}^*=\Ad_h^*\circ \Ad_g^*.
\]
This yields the required group homomorphism.
\end{proof}

Note that if $V$ is a finite-dimensional vector space, so is the space $\End(V)$ of linear maps $V\to V$, so we may regard $\End(V)$ as a manifold. Moreover, for any finite-dimensional vector space $W$, the tangent space $T_wW$ at any point $w:W$ is canonically identified with $W$ itself. In particular, $\End(\lie{g})$ and $\End(\lie{g}^*)$ are smooth manifolds, and a tangent vector to either one is canonically an element of the same vector space. 

Hence the assignments
\[
\Ad\colon G\to\End(\lie{g})
\qqand
\Ad^*\colon G\to\End(\lie{g}^*)
\]
given by $g\mapsto \Ad_g$ and $g\mapsto \Ad_g^*$ are smooth maps.
Differentiating $\Ad$ and $\Ad^*$ at the identity gives
\[
\ad\colon \lie{g}\to \End(\lie{g}),
\qqand
\ad^*\colon \lie{g}\to \End(\lie{g}^*).
\]
For each $X:\lie{g}$ we write
\[
\ad_X\colon \lie{g}\to \lie{g}
\qqand
\ad_X^*\colon \lie{g}^*\to \lie{g}^*
\]
for what are called the \emph{adjoint operator} and \emph{coadjoint operator} associated to $X$. Note that the transpose operation $(-)^*\colon \End(\lie{g})\to \End(\lie{g}^*)$ is linear, so differentiating $\Ad_g^*=(\Ad_g)^*$ at the identity gives $\ad_X^*=(\ad_X)^*$.

The Lie bracket on $\lie{g}$ is defined by
\[
[X,Y]\coloneqq \ad_X(Y),
\]
which makes $\lie{g}$ into a Lie algebra. However, because we use brackets $[p,q]$ for internal hom of polynomials and have little use for Lie brackets in the paper, we will never again use the bracket notation for Lie algebras, using $\ad$ instead. 

Define a map
\[
\exp_G\colon \lie{g}\to G
\]
by setting $\exp_G(X)\coloneqq \gamma_X(1)$, where $\gamma_X\colon \rr\to G$ is the unique integral curve through $e$ of the left-invariant vector field on $G$ determined by $X$; this is called the \emph{exponential map}.

\subsection{SCAN Lie groups}\label{sec:scan}

For any Lie group $G$, we've seen in \cref{prop:cot-hopf} that $\cot(G)$ is a Hopf monoid. In fact, it is sometimes a $\tri$-comonoid as well, i.e.\ a category; \cref{rmk:scan-fiberwise-addition} explains where this category structure on $\cot(G)$ comes from in the SCAN case.

\begin{definition}\label{def:scn}
A \emph{simply connected nilpotent} (SCN) Lie group, or simply a \emph{SCN}, is a connected, simply connected, finite-dimensional real Lie group for which there exists $n:\nn$ such that, for all $X_1,\ldots,X_n:\lie{g}$, the iterated composite of adjoint operators is zero:
\[
\ad_{X_1}\circ\ad_{X_2}\circ\cdots\circ\ad_{X_n}=0\colon \lie{g}\to\lie{g}.
\]
The least such $n$ is called the \emph{nilpotency class} of $\lie{g}$.
\end{definition}

Note that $\lie{g}$ has nilpotency class $n$ iff $\lie{g}^*$ does.

\begin{lemma}[Malcev]\label{lem:malcev}
For any SCN $G$, the exponential map $\exp_G\colon\lie{g} \to G$ is a diffeomorphism.
\end{lemma}

\begin{proof}
Corwin--Greenleaf~\cite[Theorem 1.2.1]{corwin1990representations}.
\end{proof}

\begin{definition}\label{def:scan}
A \emph{simply connected anchored nilpotent} (SCAN) Lie group, or simply a \emph{SCAN}, is a pair $(G, \sharp_G)$ where $G$ is a SCN and $\sharp_G\colon\lie{g}^* \to \lie{g}$ is a chosen linear isomorphism, called the \emph{anchor}. We write $\scan$ for the groupoid whose objects are SCANs and whose morphisms $(G,\sharp_G)\to(H,\sharp_H)$ are Lie-group isomorphisms $\phi\colon G\xrightarrow{\cong}H$,%
\footnote{We use isomorphisms here for technical simplicity later; see \cref{some note on what would have happened if we'd used arb morphisms}.}
whose pushforward $\phi_*\coloneqq d\phi|_e\colon\lie{g}\to\lie{h}$ and pullback $\phi^*\colon\lie{h}^*\to\lie{g}^*$ make the following square commute:
\[
\begin{tikzcd}
\lie{h}^* \ar[r,"\sharp_H"] \ar[d,"\phi^*"'] & \lie{h} \\
\lie{g}^* \ar[r,"\sharp_G"'] & \lie{g} \ar[u,"\phi_*"']
\end{tikzcd}
\qedhere\]
\end{definition}

\begin{remark}\label{rmk:anchor-chosen}
In Lie theory, a map $\lie{g}^*\to\lie{g}$ of the above sort is usually just regarded as a chosen identification of $\lie{g}^*$ with $\lie{g}$, often arising from a nondegenerate bilinear form $\lie{g}\times\lie{g}\to\rr$. We call $\sharp_G$ an \emph{anchor} by analogy with the anchor maps of Poisson geometry and Lie algebroid theory, where an anchor is a bundle map to $TG$; the usual Poisson anchor need not be an isomorphism, but the present setting forces it: $\sharp_G$ is required to be a linear isomorphism, providing an identification of $\lie{g}^*$ with $\lie{g}$. The anchor is part of the data of a SCAN; distinct choices of $\sharp_G$ give distinct SCAN structures on the same underlying SCN. \Cref{prop:canonical-anchor} below produces a canonical anchor for cotangent bundles of SCNs, but this is one valid choice among many.
\end{remark}

\begin{example}\label{ex:Rn-scan}
The additive group $(\mathbb{R}^n,+)$ is a SCN Lie group for any $n:\nn$: abelian, hence nilpotent of class~$1$; simply connected and contractible. Its Lie algebra is $\lie{g}=T_0\mathbb{R}^n=\mathbb{R}^n$, the exponential map $\exp_{\mathbb{R}^n}\colon\mathbb{R}^n\to\mathbb{R}^n$ is the identity, and left translation $L_p\colon q\mapsto p+q$ has differential the identity, so $(dL_p|_e)^*=\id$ at every $p$. 

Any linear isomorphism $\sharp\colon(\mathbb{R}^n)^*\to\mathbb{R}^n$ can serve as an anchor. Under the resulting SCAN structure $(\mathbb{R}^n,\sharp)$, the on-directions bijection $F_p\colon T^*_p\mathbb{R}^n\to\mathbb{R}^n$ of \cref{prop:scan-polynomial} reduces to $\sharp$, and the induced codomain map on $\cot(\mathbb{R}^n)\cong\mathbb{R}^n\yon^{\mathbb{R}^n}$ sends $\xi:T^*_p\mathbb{R}^n$ to $p+\sharp(\xi)$, since group multiplication on $\mathbb{R}^n$ is addition.
\end{example}

\begin{proposition}\label{prop:scan-smc-action}
The groupoid $\scan$ inherits a symmetric monoidal structure $(\times,\rr^0)$, and the product functor
\[
(\cdot)\colon\scan\times\mfd\to\mfd,\qquad (G,M)\mapsto G\times M
\]
is a strong symmetric monoidal action of $\scan$ on $\mfd$.
\end{proposition}

\begin{proof}
If $(G_1,\sharp_1)$ and $(G_2,\sharp_2)$ are SCAN, then $G_1\times G_2$ is connected and simply connected, and its Lie algebra $\lie{g}_1\oplus\lie{g}_2$ has nilpotency class $\max(c_1,c_2)$ where $c_i$ is the class of $\lie{g}_i$ [proof or reference?]; the direct sum $\sharp_1\oplus\sharp_2\colon(\lie{g}_1\oplus\lie{g}_2)^*\to\lie{g}_1\oplus\lie{g}_2$ is a linear isomorphism, hence a SCAN anchor. The zero-dimensional group $\rr^0$ is SCAN, serving as the monoidal unit. The cartesian product of Lie-group isomorphisms is a Lie-group isomorphism, and intertwining with anchors is preserved componentwise, so $\times$ restricts to a functor $\scan\times\scan\to\scan$; the associator, unitors, and braiding of $\mfd$ are Lie-group isomorphisms between SCANs compatible with the direct-sum anchors, giving $(\scan,\times,\rr^0)$ the structure of a symmetric monoidal groupoid. Note that $\times$ is \emph{not} a categorical product in $\scan$: the projections $G\times H\to G$ are not Lie-group isomorphisms in general. The action $(\cdot)$ extends $\times\colon\scan\times\scan\to\scan$ along the forgetful functor $\scan\to\mfd$ (on objects $G\mapsto G$, on morphisms treat each SCAN iso as a diffeomorphism), and since $\times\colon\mfd\times\mfd\to\mfd$ is strong symmetric monoidal, so is its restriction $(\cdot)$.
\end{proof}

\begin{proposition}\label{prop:scan-polynomial}
For $(G, \sharp_G)$ SCAN, there is an isomorphism $\cot(G) \cong G \yon^G$ in $\poly$.
\end{proposition}

\begin{proof}
Positions on both sides are $G$. For each $g:G$, compose bijections
\begin{equation}\label{eqn.directions_malcev}
T^*_g G \xrightarrow{(dL_g|_e)^*} \lie{g}^* \xrightarrow{\sharp_G} \lie{g} \xrightarrow{\exp_G} G,
\end{equation}
using that $(dL_g|_e)^*\colon T^*_g G \to \lie{g}^*$ is a linear iso (inverse of the transpose of left translation), $\sharp_G$ is a linear iso by \cref{def:scan}, and $\exp_G$ is a bijection by \cref{lem:malcev}.
\end{proof}

\begin{remark}\label{rmk:scan-fiberwise-addition}
For a SCAN $(G,\sharp_G)$ and point $g:G$, the on-directions part of the isomorphism $\cot(G)\cong G\yon^G$ from \cref{prop:scan-polynomial} is the bijection \eqref{eqn.directions_malcev}, which we'll temporarily denote $F_g\colon T^*_gG\to G.$

By \cref{prop:comonads-cats}, a $\tri$-comonoid structure on $G\yon^G$ is a category whose object set is $G$ and whose arrows out of each object are indexed by $G$. We describe such a category, the \emph{translation category} of $G$ acting on itself by left multiplication: objects are elements of $G$, a morphism $g\to h$ is an element $x:G$ satisfying $h=gx$ (there is exactly one such $x$, namely $g\inv h$), identities are $e:G$, and composition is group multiplication. This category is isomorphic to the pair groupoid on $G$, but the translation-category presentation is the one that matches \eqref{eqn.directions_malcev}: the bijection $F_g\colon T^*_gG\to G$ identifies a cotangent vector at $g$ with a morphism $g\to gF_g(\xi)$ in the translation category. Thus we have maps
\[
\epsilon\colon\cot(G)\to\yon
\qqand
\delta\colon\cot(G)\to\cot(G)\tri\cot(G).
\]
The former, which provides identities, picks out the zero covector at each point of $G$, because $F_g(0)=\exp_G(0)=e$. The latter induces codomain and composition maps on cotangent vectors as follows. A covector $\xi:T^*_gG$ is sent to the point
\[
\cod_g(\xi)\coloneqq gF_g(\xi):G,
\]
and for $\xi':T^*_{\cod_g(\xi)}G$, the composite of $\xi$ with $\xi'$ is determined by group multiplication:
\[
F_g(\xi\then\xi') =F_g(\xi)\,F_{\cod_g(\xi)}(\xi').
\]
Thus the chosen anchor $\sharp_G$, together with $\exp_G$, identifies each cotangent space with $G$ in a way that presents the induced category structure on $\cot(G)$ as the translation category of $G$ acting on itself by left multiplication, hence as a category isomorphic to the pair groupoid. For those who know the Baker--Campbell--Hausdorff formula on SCNs, after transporting further along $\sharp_G\colon\lie{g}^*\to\lie{g}$, the composition law on $\cot(G)$ is given by the BCH product.
\end{remark}

\begin{proposition}\label{prop:TstarG-scn}
For any SCN $G$, the cotangent bundle manifold $T^*G$ is a SCN under left-trivialization.
\end{proposition}

\begin{proof}
Left-trivialize $T^*G \cong G \times \lie{g}^*$ via $(g, \xi) \leftrightarrow (g, (dL_g|_e)^*\xi)$ as in \eqref{eqn.left_triv}. Give $G \times \lie{g}^*$ the operation
\[
(g_1, \xi_1) \cdot (g_2, \xi_2) := (g_1 g_2, \; \xi_1 + (\Ad^*_{g_1\inv}\xi_2)),
\]
with unit $(e, 0)$ and inverse $(g, \xi)\inv = (g\inv, -\Ad^*_g \xi)$. The unit and associativity both follow from \cref{lem:Ad-facts}; in particular,
\[
\Ad_e^*=\id_{\lie{g}^*}
\qqand
\Ad^*_{g_2\inv}\circ\Ad^*_{g_1\inv}=\Ad^*_{(g_2g_1)\inv}.
\]
The cotangent bundle is well-known to be a smooth manifold. Since $G$ is simply connected and $\lie{g}^*$ is contractible, the product is simply connected. It remains to show nilpotency.

The Lie algebra of $G\times\lie{g}^*$ is its tangent space at the identity element $(e,0)$, namely
\[
T_{(e,0)}(G\times\lie{g}^*) \cong T_eG\oplus T_0(\lie{g}^*)=\lie{g}\oplus\lie{g}^*.
\]
Its adjoint action is
\[
\ad_{(X_1,\nu_1)}(X_2,\nu_2)=\big(\ad_{X_1}(X_2), \; \ad^*_{X_2}\nu_1 - \ad^*_{X_1}\nu_2\big),
\]
obtained by differentiating the product law. Relative to the decomposition $\lie{g}\oplus\lie{g}^*$, each operator $\ad_{(X,\nu)}$ has block form
\[
\ad_{(X,\nu)}=
\begin{bmatrix}
\ad_X & 0\\
* & -\ad_X^*
\end{bmatrix}.
\]
Suppose $\lie{g}$ has nilpotency class $n$, so every $n$-fold composite of operators $\ad_X$ is zero on $\lie{g}$; transposing, every $n$-fold composite of operators $\ad_X^*$ is zero on $\lie{g}^*$. We show that any $(2n-1)$-fold composite $A_k\cdots A_1$ (where $A_i=\ad_{(X_i,\nu_i)}$) is zero.

Write $A_i=\begin{bsmallmatrix}\ad_{X_i} & 0\\ *_i & -\ad_{X_i}^*\end{bsmallmatrix}$. A product of lower-triangular block matrices is lower-triangular, with diagonal blocks the products of diagonal blocks. So the $(1,1)$ entry of $A_k\cdots A_1$ is $\ad_{X_k}\cdots\ad_{X_1}$, which is zero for $k\geq n$; similarly for the $(2,2)$ entry $(-1)^k\ad^*_{X_k}\cdots\ad^*_{X_1}$.

For the $(2,1)$ entry: expanding the matrix product, each term in the $(2,1)$ entry of $A_k\cdots A_1$ is a product
\[
D_k^{(2,2)}\cdots D_{j+1}^{(2,2)}\cdot *_j\cdot D_{j-1}^{(1,1)}\cdots D_1^{(1,1)}
\]
for some choice of $j:\{1,\ldots,k\}$, where $D_i^{(1,1)}=\ad_{X_i}$ and $D_i^{(2,2)}=-\ad^*_{X_i}$. When $k=2n-1$, either $j-1\geq n$ (the right factor contains an $n$-fold $\ad$-composite and vanishes) or $k-j\geq n$ (the left factor contains an $n$-fold $\ad^*$-composite and vanishes). So every term is zero, and $\lie{g}\oplus\lie{g}^*$ has nilpotency class at most $2n-1$.
\end{proof}

\begin{proposition}\label{prop:canonical-anchor}
The canonical symplectic form $\omega$ on $T^*G$ induces a canonical anchor $\sharp_{T^*G}\colon (\lie{g} \oplus \lie{g}^*)^* \to \lie{g} \oplus \lie{g}^*$, giving $T^*G$ a canonical SCAN structure.
\end{proposition}

\begin{proof}
This is standard; see Abraham--Marsden~\cite[\S3.2]{abraham1978foundations} or Marsden--Ratiu~\cite[Prop 6.1.1]{marsden1999introduction}. The canonical symplectic form on $T^*G$ is nondegenerate, so at $(e,0)$ it induces an isomorphism
\[
\flat_\omega\colon \lie{g}\oplus\lie{g}^* \to (\lie{g}\oplus\lie{g}^*)^*,
\]
whose inverse is the required anchor $\sharp_{T^*G}$.
\end{proof}

\section{Dynamic category-like structures}

Categories, monoidal categories, traced categories, operads, etc.\ can be understood in terms of the sorts of compositions they allow. For example:
\begin{equation}\label{eqn.wd_examples}
\begin{tikzpicture}[baseline=(Catg)]
\begin{scope}[font=\footnotesize, text height=1.5ex, text depth=.5ex]
  \begin{scope}[oriented WD, bb port sep=1, bb port length=0, bb min width=.4cm, bby=.2cm, inner xsep=.2cm, x=.5cm, y=.3cm]

\noindent In each case there is an associated operad $\cat{W}$ whose objects are the legal interfaces (boxes-with-objet-labeled-ports) and whose morphisms $\phi\colon b_1,\ldots,b_K\to c$ are the legal arrangements of boxes in a box. That is, there are operads $\cat{W}_{\tn{O-Cat}}$, $\cat{W}_{\tn{O-MnCat}}$, $\cat{W}_{\tn{O-TrCat}}$, and $\cat{W}_{\tn{O-Opd}}$ each with a 2-ary morphism that can be drawn as above.%
\footnote{
There is a potentially-confusing level-shift in that there is an operad $\cat{W}_{\tn{O-Opd}}$ for which algebras are operads with object set $O$. In case it's helpful, the object set for $\cat{W}_{\tn{O-Opd}}$ would be $\List(O)\times O$, because every ``box'' has a list of inputs and an output.
}

A $\cat{W}$-algebra $\cat{C}\colon\cat{W}\to\smset$ assigns to each box $b$ the set $\cat{C}(b)$ of ``fillers'' and to each arrangement $\phi$ the function $\cat{C}(\phi)\colon \cat{C}(b_1)\times\cdots\times \cat{C}(b_K)\to \cat{C}(c)$ sending fillers for little boxes to a filler of the big box.%
\footnote{By a $\cat{W}$-algebra, we mean an operad functor $\cat{W}\to\smset$, where $\smset$ is the operad (multicategory) underlying the monoidal category $(\smset,1,\times)$.}

\begin{example}
For a set $O$, let $\ob\cat{W}_{\tn{O-Cat}}\coloneqq O\times O$ and
\[\cat{W}_{\tn{O-Cat}}((o_0',o_1),(o_1',o_2),\ldots(o_{K-1}',o_K);(o_0,o_K'))\coloneqq
\begin{cases}
\{1\}&\tn{ if }o_k=o_k'\quad \forall\; 0\leq k\leq K\\
\varnothing&\tn{ otherwise}
\end{cases}
\]
Thus the first diagram shown in \eqref{eqn.wd_examples} represents a map $\phi\colon (A,B),(B,C)\to(A,C)$ in $\cat{W}_{\tn{O-Cat}}$ for some $O$ with $A,B,C:O$. If $\cat{C}\colon\cat{W}_{\tn{O-Cat}}\to\smset$ is an algebra, then the above diagram shows $f_1:\cat{C}(A,B)$, $f_2:\cat{C}(B,C)$ and $\phi(f_1,f_2):\cat{C}(A,C)$. In other words the diagram $\phi$ represents composition in a category. The category of small categories with object set $O$ is equivalent to the category of $\cat{W}_{\tn{O-Cat}}$-algebras.
\end{example}

A variety of examples of these ideas---wiring-diagram operads whose algebras are structured categories---can be found in the literature. \cite{Spivak.Schultz.Rupel:2016a} treats the traced and compact cases, showing in particular that the operad $\cat{W}_{\tn{TrCat}}$ for traced categories is the operad $1\tn{-}\Cat{Cob}$ of oriented 1-dimensional cobordisms. \cite{fong2019hypergraph} treats hypergraph categories, whose associated operad $\cat{W}_{\tn{HypCat}}$ is given by cospans of finite sets. The symmetric monoidal case~\cite{patterson2021wiring} is in some sense the most basic of the three, but its operad $\cat{W}_{\tn{MnCat}}$ is less easy to describe directly.

An advantage of the operadic perspective is that it allows us to more uniformly treat various sorts of category, including ones that may not have an existing name. For example, 
consider the notion of ``traced monoidal categories without identity''; one might balk at this because they wouldn't know whether the ``yanking'' axiom could so easily just be discarded. But as an operad it is fine: it is the suboperad comprising the left class of a certain factorization system on $1\tn{-}\Cat{Cob}$.

Another advantage is that wiring diagram algebras also easily represent $\cat{V}$-enriched categories for monoidal categories $\cat{V}$, namely by using algebras $\cat{W}\to\cat{V}$ valued in $\cat{V}$. For example, a $\Cat{Vect}$-enriched category with object set $O$ is an opeard functor $\cat{W}_{\tn{O-Cat}}\to\Cat{Vect}$.


One can build \emph{dynamic} versions of the structures in \eqref{eqn.wd_examples}, in which the ``fillers'' of a box are ``morphisms that change in time'', i.e.\ coalgebras.  In \cite{shapiro2022dynamic}, a dynamic category is a category enriched in $\org$; similarly for dynamic monoidal categories or dynamic operads. Given the ideas above, these are just operad functors
\[
\cat{W}_{\tn{O-Cat}}\to\org
\qquad
\cat{W}_{\tn{O-MnCat}}\to\org
\qquad
\cat{W}_{\tn{O-Opd}}\to\org
\] 
for some $O$. Our next goal then is to recall $\org$.

\subsection{The monoidal bicategory $\org$}\label{sec:org}

We recall the monoidal bicategory $\org$ from~\cite{spivak2021learners}; see also~\cite{shapiro2022dynamic}. The objects of $\org$ are polynomials, and for polynomials $p,q$ the hom-category is
\[\org(p,q)\coloneqq\ihom{p,q}\tn{-}\Cat{Coalg}.\]
Its monoidal structure is induced by the tensor product of polynomials. A morphism $p\to q$ in $\org$ is an $\ihom{p,q}$-coalgebra $(S,\,\beta\colon S\to\ihom{p,q}(S))$, specifying for each $s:S$ an action $\act(s):\poly(p,q)$ and an update $\upd_s\colon\ihom{p,q}[\act(s)]\to S$ assigning a successor state to each direction.

\begin{definition}\label{def:pq-coalg}
The category $\ihom{p,q}\coalg$ has as objects pairs $(S,\beta)$ where $S$ is a set and $\beta\colon S\to\ihom{p,q}\tri S$ is a function, and where a morphism from $(S,\beta)$ to $(S',\beta')$ is a function $f\colon S\to S'$ such that $(\ihom{p,q}\tri f)\circ\beta=\beta'\circ f$. We refer to $S$ as the \emph{state set}.
\end{definition}

Unwinding this using \eqref{eqn:dirichlet}, $\beta=(\act^\beta,\upd^\beta)$ decomposes as an \emph{action} function
\[\act^\beta\colon S\to\poly(p,q)=\ihom{p,q}(1)\]
and, for each state $s:S$, an \emph{update} function
\[\upd^\beta_s\colon\sum_{I:p(1)}q\left[\act^\beta_s(I)\right]\to S.\]
For a state $s:S$ and position $I:p(1)$ we often write $\act^\beta_s\colon p\to q$ and $\upd^\beta_s(I)\colon q[\act^\beta_s(I)]\to S$. We may suppress $\beta$ when clear from context.

\begin{proposition}[\cite{spivak2021learners,shapiro2022dynamic}]\label{prop:org}
The hom-categories $\ihom{p,q}\coalg$ assemble into a monoidal bicategory $\org$ with polynomials as objects and tensor product induced by the tensor product of polynomials.
\end{proposition}

This structure arises from the fact that $\coalg\colon(\poly,\yon,\otimes)\to(\smcat,1,\times)$ is lax monoidal:
\[
1\to\yon\coalg
\qqand
p\coalg\times q\coalg\to(p\otimes q)\coalg.
\]
Thus the structure maps $\yon\to[p,p]$ and $[p,q]\otimes[q,r]\to[p,r]$ induce the identity and composition functors
\[
1\to[p,p]\coalg
\qqand
[p,q]\coalg\times[q,r]\coalg\to[p,r]\coalg.
\]

\subsection{Example: deep learning as a dynamic monoidal category}\label{sec:deep-learning}

We recast the deep-learning example of~\cite{shapiro2022dynamic} using the cotangent functor and SCAN Lie groups. Take the object set $O\coloneqq\nn$ and assign to each $n:\nn$ the polynomial $\cot(\mathbb{R}^n)\cong\cot(\mathbb{R})\tpow n$; a layer $m\to n$ will be an $\ihom{\cot(\mathbb{R}^m),\cot(\mathbb{R}^n)}$-coalgebra.

The data of a layer is a parameter space $P=\mathbb{R}^k$, regarded as a SCAN Lie group with anchor $\sharp_P=-\eta\cdot\id$ for a learning rate $\eta>0$ (\cref{ex:Rn-scan}), together with a smooth architecture $f\colon P\times\mathbb{R}^m\to\mathbb{R}^n$. Applying $\cot$ and using its strong monoidality (\cref{prop:cot-monoidal}) gives a polynomial map
\[\cot(P)\otimes\cot(\mathbb{R}^m)\to\cot(\mathbb{R}^n),\]
which by closedness of $\poly$ is equivalently a map $\cot(P)\to\ihom{\cot(\mathbb{R}^m),\cot(\mathbb{R}^n)}$. By \cref{ex:Rn-scan}, $\cot(P)\cong P\yon^P$, so \cref{ex:coalg-as-poly-map} presents it as an $\ihom{\cot(\mathbb{R}^m),\cot(\mathbb{R}^n)}$-coalgebra with state set $P$.

Unwinding: the action at $p:P$ is $\cot(f_p)\colon\cot(\mathbb{R}^m)\to\cot(\mathbb{R}^n)$ (forward part $x\mapsto f_p(x)$, backward part $(Df_p)_x^\top$), and the update at $p$ on direction $(x,\nu):\mathbb{R}^m\times T^*_{f_p(x)}\mathbb{R}^n$ is
\[
\upd_p(x,\nu)=p+\sharp_P\bigl((\partial_Pf)_{(p,x)}^\top\nu\bigr)=p-\eta\cdot(\partial_Pf)_{(p,x)}^\top\nu,
\]
using \cref{ex:Rn-scan}'s codomain map on $\cot(P)\cong P\yon^P$. When $\nu$ is the gradient of a loss $L$ at $f_p(x)$, the chain rule gives $(\partial_Pf)_{(p,x)}^\top\nu=\nabla_pL$, so $\upd_p$ is one step of stochastic gradient descent.

Composition in $\org$ reproduces backpropagation; parallel composition of layers corresponds via \cref{prop:cot-monoidal} to $\cot(\mathbb{R}^m)\otimes\cot(\mathbb{R}^{m'})\cong\cot(\mathbb{R}^{m+m'})$, and the whole assembles into an operad functor
\[\cat{L}\colon\cat{W}_{\nn\tn{-MnCat}}\to\org.\]

\subsection{Scan-parameterized manifold lenses}

We next add the parameterization using the $\Para$ construction, introduced in~\cite[Section 2]{hermida2012monoidal} and further developed in~\cite{fong2019backprop,capucci2024actegories}. Given a symmetric monoidal category $(\cat{A},I,\otimes)$ and a monoidal action
\[
(\cdot)\colon\cat{A}\times\cat{M}\to\cat{M}
\]
the category $\para{\cat{A}}{\cat{M}}$ is defined to have the same objects as $\cat{M}$, and a morphism $M_1\to M_2$ is a pair $(P,\ell)$ where $P\in\cat{A}$ and $\ell\colon P\cdot M_1\to M_2$ in $\cat{M}$. Identity uses the unit $I$ as parameter in the evident way, and composition is given by
\[
(P_2,\ell_2)\circ (P_1,\ell_1)\coloneqq\bigl(P_1\otimes P_2,\;\ell_2\circ(\id_{P_2}\cdot\ell_1)\bigr).
\]
When $\cat{M}$ is itself symmetric monoidal and the action $(\cdot)$ is strong symmetric monoidal, $\para{\cat{A}}{\cat{M}}$ inherits a symmetric monoidal structure in an obvious way.

In our setting, $\cat{A}\coloneqq\scan$ acts on $\cat{M}\coloneqq\lmfd$ by the lift of \cref{prop:scan-smc-action}'s product action to the $\outp M$-slot:
\[
(\cdot)\colon\scan\times\lmfd\to\lmfd,\qquad Q\cdot(\inpt M,\outp M)\coloneqq(\inpt M,Q\times\outp M).
\]
We abbreviate $\plm\coloneqq\para{\scan}{\lmfd}$ (parameterized lens of manifolds). A morphism $g\colon(\inpt M,\outp M)\to(\inpt N,\outp N)$ in $\plm$ is a tuple $g=(Q,\outp g,\inpt g)$ with $Q$ a SCAN Lie group and

\inpt g\colon Q\times\outp M\times\inpt N\to\inpt M.

\subsection{Potentialized lenses}
Finally we add potentials to $\lmfd$, defining a category $\potlens$ as follows. Its objects are the same, $\ob\potlens\coloneqq\ob\lmfd$. \end{align*}
